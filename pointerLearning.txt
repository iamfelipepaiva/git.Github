#include <stdio.h>
char letterCounting(char *sentence);


int main(){
    char whatever[50];
    printf("Insert whatever info you may want: ");
    scanf("%[^\n]", whatever);
    printf("Number of input characters: %i", letterCounting(whatever));

    return 0;
}

char letterCounting(char *sentence){
    char *ptoStarter = sentence;
    while(*sentence != '\0'){
        sentence++;
    }

    int sizeOfIt = sentence - ptoStarter;

    return sizeOfIt;
}
